Q.1 - Problem:
You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string. 
                Input:abcadeecfb
                Output:abcdef

Ans - 
                function removeDuplicates(str) {
                    const charObj = {};
                    let result = '';
                    for (let char of str) {
                      if (!charObj[char]) {
                        charObj[char] = true;
                        result += char;
                      }
                    }
                    return result;
                  }
                  const input = 'abcadeecfb';
                  const output = removeDuplicates(input);
                  console.log(output);

Q.2 - Problem:
You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) 
                    Input: abcadeecfb

                    Output:
                            a=2
                            b=2
                            c=2
                            d=1
                            e=2
                            f=1"

Ans - 
                let a = 'abcadeecfb';
                function countAlpha(str) {
                  let alphabetMap = new Map();
                  for (let char of str) {                    
                    if (alphabetMap.has(char)) {
                      alphabetMap.set(char, alphabetMap.get(char) + 1);
                    }                    
                    else {
                      alphabetMap.set(char, 1);
                    }
                  }                  
                  for (let [key, value] of alphabetMap) {
                    console.log(key + '=' + value);
                  }
                }
                countAlpha(a);
